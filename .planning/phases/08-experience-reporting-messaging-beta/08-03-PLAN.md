IPF-03 pass-rate formula (locked via 08-01 schema):

```sql
select
  (count(*) filter (where sc.is_faa_checkride and sc.status='passed' and sc.attempt_number=1)::numeric
   / nullif(count(*) filter (where sc.is_faa_checkride and sc.attempt_number=1), 0)) as pass_rate_first_attempt,
  count(*) filter (where sc.is_faa_checkride and sc.attempt_number=1) as attempts_total,
  count(*) filter (where sc.is_faa_checkride and sc.status='passed' and sc.attempt_number=1) as passes_first_attempt
from public.stage_check sc
where sc.instructor_id = $1
  and sc.deleted_at is null
  and sc.school_id = (auth.jwt() ->> 'school_id')::uuid
```

    ```typescript
    export async function instructorDutyMinutesInWindow(
      tx: PgTransaction, instructorId: string, windowEnd: Date,
    ): Promise<number> {
      const rows = await tx.execute(sql`
        select coalesce(sum(
          extract(epoch from (upper(time_range) - lower(time_range))) / 60
        ), 0)::int as minutes
          from public.reservation
         where instructor_id = ${instructorId}::uuid
           and deleted_at is null
           and activity_type = 'flight'
           and status in ('approved','dispatched','flown','pending_sign_off','closed')
           and lower(time_range) >= ${windowEnd.toISOString()}::timestamptz - interval '24 hours'
           and lower(time_range) <  ${windowEnd.toISOString()}::timestamptz
      `);
      return rows[0]?.minutes ?? 0;
    }

    export async function checkDutyHoursForProposal(
      tx: PgTransaction,
      { instructorId, proposedStart, proposedEnd }: Args,
    ): Promise<Result> {
      if (proposedStart <= new Date()) {
        return { warn: false, block: false, existingMinutes: 0, proposedMinutes: 0, totalMinutes: 0 };
      }
      const existingMinutes = await instructorDutyMinutesInWindow(tx, instructorId, proposedEnd);
      const proposedMinutes = Math.ceil((proposedEnd.getTime() - proposedStart.getTime()) / 60_000);
      const totalMinutes = existingMinutes + proposedMinutes;
      return {
        warn: totalMinutes > 7 * 60,
        block: totalMinutes > 8 * 60,
        existingMinutes, proposedMinutes, totalMinutes,
      };
    }
    ```
