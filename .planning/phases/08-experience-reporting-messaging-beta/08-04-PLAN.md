export async function GET(req: Request) {
  const caller = await resolveCallerContext(); // resolves session + role

  const { searchParams } = new URL(req.url);

  const caller2 = appRouter.createCaller(ctx); // reuse tRPC procedure
  const rows = await caller2.reports.fleetUtilization.query({ baseId, from, to });

  const csv = [
    'tail_number,make_model,flight_hours,utilization_pct,squawk_count',
    ...rows.map((r) =>
      [
        escapeCsv(r.tail),
        escapeCsv(r.makeModel),

Report scope helper contract:

```typescript
export async function resolveReportScope({ baseId, caller }: Args): Promise<{
  baseIdsFilter: string[] | null; // null = all bases (admin only); otherwise the one(s) to include
  isRollup: boolean;
}> {
  if (caller.role !== 'admin' && (baseId === 'all' || baseId !== caller.activeBaseId)) {
    throw new TRPCError({
      code: 'FORBIDDEN',
      message: 'Only admins can view reports across bases',
    });
  }
  if (baseId === 'all' && caller.role === 'admin') return { baseIdsFilter: null, isRollup: true };
  if (baseId) return { baseIdsFilter: [baseId], isRollup: false };
  return { baseIdsFilter: [caller.activeBaseId], isRollup: false };
}

export function resolveDateBoundaries({
  from,
  to,
  timezone,
}: {
  from: string;
  to: string;
  timezone: string;
}): { fromUtc: Date; toUtc: Date } {
  // Use date-fns-tz zonedTimeToUtc
  return {
    fromUtc: zonedTimeToUtc(from + 'T00:00:00', timezone),
    toUtc: zonedTimeToUtc(to + 'T23:59:59.999', timezone),
  };
}
```
