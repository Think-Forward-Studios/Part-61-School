      return t.middleware(({ ctx, next }) => {
        if (!ctx.session) throw new TRPCError({ code: 'UNAUTHORIZED' });
        if (!allowed.includes(ctx.session.activeRole)) {
          throw new TRPCError({ code: 'FORBIDDEN', message: `Requires one of: ${allowed.join(', ')}` });

        return next({ ctx });

    Create `apps/web/app/(app)/switch-role/actions.ts` as a Server Action:
    ```ts
    'use server';
    import { cookies } from 'next/headers';
    import { redirect } from 'next/navigation';
    import { createSupabaseServerClient } from '@/lib/supabase/server';

    const ALLOWED = ['student','instructor','mechanic','admin'] as const;
    type Role = typeof ALLOWED[number];

    export async function switchRole(formData: FormData) {
      const role = formData.get('role') as Role;
      if (!ALLOWED.includes(role)) throw new Error('Invalid role');

      const supabase = await createSupabaseServerClient();
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error('Not signed in');

      // Validate the role is in the user's JWT roles[] claim (defense-in-depth)
      const { data: session } = await supabase.auth.getSession();
      const jwt = session.session?.access_token;
      // Decode JWT payload (middle section, base64url) — or call a tRPC me() endpoint
      const payload = JSON.parse(Buffer.from(jwt!.split('.')[1], 'base64url').toString());
      if (!Array.isArray(payload.roles) || !payload.roles.includes(role)) {
        throw new Error('User does not hold that role');
      }

      const cookieStore = await cookies();
      cookieStore.set('part61.active_role', role, { httpOnly: true, sameSite: 'lax', secure: process.env.NODE_ENV === 'production', path: '/' });
      redirect('/');
    }
    ```
