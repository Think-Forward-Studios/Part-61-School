**/schedule/request/page.tsx** extension:
When both student and lesson are selected in the existing form (from Phase 3), add:

```tsx
const eligibility = trpc.schedule.evaluateLessonEligibility.useQuery(
  {
    enrollmentId,
    lessonId,
    aircraftId,
    instructorUserId,
  },
  { enabled: !!enrollmentId && !!lessonId },
);

{
  eligibility.data && !eligibility.data.ok && (
    <BlockerList
      blockers={eligibility.data.blockers}
      canGrantOverride={isAdminOrChiefInstructor}
      onGrantClick={() => setOverrideDialogOpen(true)}
    />
  );
}

{
  eligibility.data?.override_active && (
    <Alert>Active chief-instructor override in place for this lesson.</Alert>
  );
}

<GrantOverrideDialog
  open={overrideDialogOpen}
  onOpenChange={setOverrideDialogOpen}
  enrollmentId={enrollmentId}
  lessonId={lessonId}
  blockers={eligibility.data?.blockers ?? []}
/>;
```
